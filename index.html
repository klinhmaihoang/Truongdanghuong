const TG_TOKEN = 'YOUR_BOT_TOKEN'; const CHAT_ID = 'YOUR_CHAT_ID'; function doPost(e) { try { // Nhận từ FormData (payload field) hoặc JSON body const raw = (e.parameter && e.parameter.payload) ? e.parameter.payload : e.postData.contents; const data = JSON.parse(raw); const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = data.sheet; if (sheet === 'HocVien' && data.rows) { const sh = ss.getSheetByName('HocVien') || ss.insertSheet('HocVien'); sh.clearContents(); const headers = ['ID','Họ tên','SĐT','Loại','Khóa học','Học phí','Ngày BĐ','Trạng thái','Email','Ngày học','Đợt 1','Thời gian Đ1','Đợt 2','Thời gian Đ2','Ghi chú','Cập nhật']; sh.getRange(1, 1, 1, headers.length).setValues([headers]).setFontWeight('bold'); if (data.rows.length) { const rows = data.rows.map(s => [ s.id||'', s.name||'', s.phone||'', s.type||'', s.course||'', s.fee||0, s.start||'', s.status||'', s.email||'', s.class_date||'', s.paid1||'Chưa thu', s.paid1_at||'', s.paid2||'Chưa thu', s.paid2_at||'', s.note||'', data.time||'' ]); sh.getRange(2, 1, rows.length, headers.length).setValues(rows); } } else if (sheet === 'TonKho' && data.rows) { const sh = ss.getSheetByName('TonKho') || ss.insertSheet('TonKho'); sh.clearContents(); const headers = ['ID','Tên nguyên liệu','Tồn kho','Đơn vị','Tối thiểu','Giá nhập','Cập nhật']; sh.getRange(1,1,1,headers.length).setValues([headers]).setFontWeight('bold'); if (data.rows.length) { const rows = data.rows.map(r => [r.id||'',r.name||'',r.qty||0,r.unit||'',r.min||0,r.cost||0,data.time||'']); sh.getRange(2,1,rows.length,headers.length).setValues(rows); } } else if (sheet === 'DonHang') { const sh = ss.getSheetByName('DonHang') || ss.insertSheet('DonHang'); if (sh.getLastRow() === 0) sh.getRange(1,1,1,7).setValues([['Mã ĐH','Khách','Sản phẩm','Tổng tiền','Trạng thái','Thời gian','Ghi chú']]).setFontWeight('bold'); sh.appendRow([data.id||'',data.customer||'',data.items||'',data.total||0,data.status||'',data.time||'',data.note||'']); } else if (sheet === 'LichSuKho') { const sh = ss.getSheetByName('LichSuKho') || ss.insertSheet('LichSuKho'); if (sh.getLastRow() === 0) sh.getRange(1,1,1,5).setValues([['Thời gian','Nguyên liệu','Loại','Số lượng','Ghi chú']]).setFontWeight('bold'); sh.appendRow([data.time||'',data.name||'',data.type||'',data.qty||'',data.note||'']); } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } // Chạy tự động 8h sáng mỗi ngày — set Trigger cho hàm này function sendDailyReminders() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName('HocVien'); if (!sh || sh.getLastRow() < 2) return; const data = sh.getDataRange().getValues(); const heads = data[0]; // headers row const idx = h => heads.indexOf(h); const today = new Date(); today.setHours(0,0,0,0); const msgs = []; data.slice(1).forEach(row => { const name = row[idx('Họ tên')] || ''; const fee = +row[idx('Học phí')] || 0; const status = row[idx('Trạng thái')] || ''; const course = row[idx('Khóa học')] || ''; const class_date = row[idx('Ngày học')] || ''; const paid1 = row[idx('Đợt 1')] || ''; const paid2 = row[idx('Đợt 2')] || ''; if (status !== 'Đang học') return; const dep = Math.round(fee * 0.3); const rest = fee - dep; // Nhắc ngày học sắp tới if (class_date) { const cd = new Date(class_date); cd.setHours(0,0,0,0); const diff = Math.round((cd - today) / 86400000); if (diff === 7) msgs.push(`📅 Còn 7 ngày tới ngày học!\n👤 ${name}\n📚 ${course}\n🗓 ${class_date}\n${paid1!=='Đã thu'?'❌ Chưa cọc đợt 1: '+dep.toLocaleString()+'đ':'✅ Đã cọc đợt 1'}`); if (diff === 3) msgs.push(`⏰ Còn 3 ngày tới ngày học!\n👤 ${name}\n🗓 ${class_date}\n${paid1!=='Đã thu'?'❌ Chưa cọc: '+dep.toLocaleString()+'đ':'✅ Đã cọc'}`); if (diff === 1) msgs.push(`🔔 NGÀY MAI LÀ NGÀY HỌC!\n👤 ${name}\n📚 ${course}`); if (diff === 0) msgs.push(`🎉 HÔM NAY LÀ NGÀY HỌC!\n👤 ${name}\n📚 ${course}`); } // Nhắc chưa thanh toán đợt 1 if (fee > 0 && paid1 !== 'Đã thu') msgs.push(`💳 Chưa thanh toán đợt 1!\n👤 ${name}\n📚 ${course}\n❌ Đợt 1 (30%): ${dep.toLocaleString()}đ`); // Nhắc đợt 2 if (fee > 0 && paid1 === 'Đã thu' && paid2 !== 'Đã thu') msgs.push(`💸 Nhắc thanh toán đợt 2!\n👤 ${name}\n📚 ${course}\n✅ Đã thu đợt 1\n⏳ Đợt 2 (70%): ${rest.toLocaleString()}đ`); }); msgs.forEach(m => { UrlFetchApp.fetch(`https://api.telegram.org/bot${TG_TOKEN}/sendMessage`, { method: 'post', payload: { chat_id: CHAT_ID, text: m } }); }); }
